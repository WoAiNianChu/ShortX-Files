  "content": "// 来源: https://t.me/c/3073722829/89/1099\n\n// 根据加速度传感器获取摇一摇状态\nimportClass(android.content.Context);\nimportClass(android.hardware.Sensor);\nimportClass(android.hardware.SensorManager);\nimportClass(android.hardware.SensorEventListener);\nimportClass(android.os.Looper);\nimportClass(android.os.Handler);\nimportClass(java.util.concurrent.CountDownLatch);\nimportClass(java.util.concurrent.TimeUnit);\nimportClass(java.lang.Runnable);\n\nfunction simulateShake() {\n    var sensorManager = context.getSystemService(Context.SENSOR_SERVICE);\n    var accelerometer = sensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER);\n\n    if (accelerometer == null) {\n        return \"设备不支持加速度传感器\";\n    }\n\n    var latch = new CountDownLatch(1);\n    var isShaked = false;\n\n    var listener = new SensorEventListener({\n        onSensorChanged: function(event) {\n            var x = event.values[0];\n            var y = event.values[1];\n            var z = event.values[2];\n\n            // 计算三轴加速度的合力大小（去掉重力后相对变化）\n            var magnitude = Math.sqrt(x * x + y * y + z * z);\n\n            // 简单阈值判断：加速度变化超过 15（m/s²）认为是摇动\n            if (magnitude > 15) {\n                isShaked = true;\n                latch.countDown();\n            }\n        },\n        onAccuracyChanged: function(sensor, accuracy) {}\n    });\n\n    new Handler(Looper.getMainLooper()).post(new Runnable({\n        run: function() {\n            sensorManager.registerListener(listener, accelerometer, SensorManager.SENSOR_DELAY_GAME);\n        }\n    }));\n\n    // 最多等 3 秒\n    latch.await(3, TimeUnit.SECONDS);\n    sensorManager.unregisterListener(listener);\n\n    if (isShaked) {\n        return \"检测到摇一摇\";\n    } else {\n        return \"未检测到摇一摇\";\n    }\n}\n\nsimulateShake();\n",
